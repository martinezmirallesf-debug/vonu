function clampInt(value: unknown, min: number, max: number, fallback: number) {
  const n = Number(value);
  if (!Number.isFinite(n)) return fallback;
  return Math.max(min, Math.min(max, Math.round(n)));
}

function compactText(value: unknown, max = 1200) {
  return String(value ?? "")
    .replace(/\s+/g, " ")
    .trim()
    .slice(0, max);
}

function maskSensitiveText(value: unknown, max = 1200) {
  let text = String(value ?? "");

  text = text.replace(
    /\b[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}\b/gi,
    "[email_oculto]",
  );

  text = text.replace(
    /(\+?\d[\d\s().-]{7,}\d)/g,
    "[telefono_oculto]",
  );

  text = text.replace(
    /\b(?:\d[ -]*?){13,19}\b/g,
    "[numero_tarjeta_o_largo_oculto]",
  );

  text = text.replace(
    /\b[A-Z]{2}\d{2}[A-Z0-9]{10,30}\b/gi,
    "[iban_oculto]",
  );

  text = text.replace(
    /\b(?:\d{8}[A-Z]|[XYZ]\d{7}[A-Z])\b/gi,
    "[dni_nie_oculto]",
  );

  text = text.replace(
    /\b\d{5,8}\b/g,
    "[codigo_oculto]",
  );

  text = text.replace(
    /\b(?:seed phrase|frase semilla|private key|clave privada)\b[\s\S]{0,300}/gi,
    "[secreto_wallet_oculto]",
  );

  return compactText(text, max);
}

function safeString(value: unknown, fallback = "") {
  const s = compactText(value, 3000);
  if (!s || s.toLowerCase() === "unknown" || s.toLowerCase() === "null") {
    return fallback;
  }
  return s;
}

function safeNullableString(value: unknown) {
  const s = safeString(value, "");
  return s ? s : null;
}

function generateCaseId(report: FraudChatReport) {
  const d = new Date();
  const stamp = [
    d.getUTCFullYear(),
    String(d.getUTCMonth() + 1).padStart(2, "0"),
    String(d.getUTCDate()).padStart(2, "0"),
    String(d.getUTCHours()).padStart(2, "0"),
    String(d.getUTCMinutes()).padStart(2, "0"),
    String(d.getUTCSeconds()).padStart(2, "0"),
  ].join("");

  const shortId = String(report.id || crypto.randomUUID()).slice(0, 8);
  return `CHAT-${stamp}-${shortId}`;
}

async function callOpenAIReview(prompt: string): Promise<ReviewResult> {
  const apiKey = Deno.env.get("OPENAI_API_KEY") || Deno.env.get("openai_api_key");
  const model = Deno.env.get("OPENAI_REVIEW_MODEL") || "gpt-4o-mini";

  if (!apiKey) {
    throw new Error("Missing OPENAI_API_KEY/openai_api_key");
  }

  const response = await fetch("https://api.openai.com/v1/responses", {
    method: "POST",
    headers: {
      Authorization: `Bearer ${apiKey}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      model,
      input: prompt,
      temperature: 0.1,
      text: {
        format: {
          type: "json_object",
        },
      },
    }),
  });

  if (!response.ok) {
    const errText = await response.text();
    throw new Error(`OpenAI review error ${response.status}: ${errText}`);
  }

  const data = await response.json();
  const outputText =
    data?.output_text ||
    data?.output?.[0]?.content?.[0]?.text ||
    data?.output?.[0]?.content?.[0]?.content ||
    "";

  if (!outputText) {
    throw new Error("Empty OpenAI review response");
  }

  const parsed = JSON.parse(outputText);

  const status = String(parsed.ai_review_status || "needs_human_review") as ReviewResult["ai_review_status"];
  const allowedStatuses = new Set([
    "ai_approved",
    "ai_approved_after_redaction",
    "needs_human_review",
    "rejected",
    "duplicate_candidate",
  ]);

  const safeStatus = allowedStatuses.has(status) ? status : "needs_human_review";

  return {
    ai_review_status: safeStatus,
    ai_review_score: clampInt(parsed.ai_review_score, 0, 100, 50),
    ai_review_recommendation: safeString(parsed.ai_review_recommendation, "Revisión automática sin recomendación."),
    ai_review_reason: safeString(parsed.ai_review_reason, "Sin motivo detallado."),

    ai_duplicate_likelihood: clampInt(parsed.ai_duplicate_likelihood, 0, 100, 0),
    ai_promote_candidate:
      Boolean(parsed.ai_promote_candidate) &&
      (safeStatus === "ai_approved" || safeStatus === "ai_approved_after_redaction"),

    cleaned_case_title: maskSensitiveText(parsed.cleaned_case_title, 220),
    cleaned_case_summary: maskSensitiveText(parsed.cleaned_case_summary, 1600),
    cleaned_original_message_example: maskSensitiveText(parsed.cleaned_original_message_example, 700),
    cleaned_red_flags: maskSensitiveText(parsed.cleaned_red_flags, 1000),
    cleaned_recommended_action: maskSensitiveText(parsed.cleaned_recommended_action, 1000),
    cleaned_prevention_tips: maskSensitiveText(parsed.cleaned_prevention_tips, 1000),
    cleaned_similarity_signature: maskSensitiveText(parsed.cleaned_similarity_signature, 400),
    cleaned_keywords: maskSensitiveText(parsed.cleaned_keywords, 500),

    scam_category: safeNullableString(parsed.scam_category),
    scam_subcategory: safeNullableString(parsed.scam_subcategory),
    fraud_block: safeNullableString(parsed.fraud_block),
    channel: safeNullableString(parsed.channel),
    platform_or_app: safeNullableString(parsed.platform_or_app),
    marketplace: safeNullableString(parsed.marketplace),

async function getSimilarMatches(supabase: any, report: FraudChatReport) {
  const query = [
    report.similarity_signature,
    report.anonymized_summary,
    report.user_text_excerpt,
    report.scam_category,
    report.scam_subcategory,
  ]
    .filter(Boolean)
    .join(" ");

  if (!query.trim()) return [];

  const { data, error } = await supabase.rpc("search_fraud_reports_basic", {
    query_text: query,
    result_limit: 3,
  });

  if (error || !Array.isArray(data)) {
    console.warn("similarity_lookup_error", error);
    return [];
  }

  return data.map((m: any) => ({
    case_id: m.case_id ?? null,
    case_title: m.case_title ?? null,
    scam_category: m.scam_category ?? null,
    scam_subcategory: m.scam_subcategory ?? null,
    pattern_name: m.pattern_name ?? null,
    similarity_signature: m.similarity_signature ?? null,
    risk_score: m.risk_score ?? null,
    match_score: m.match_score ?? null,
  })) as SimilarMatch[];
}

function mapReviewStatusToChatReportStatus(finalStatus: string) {
  if (finalStatus === "ai_promoted") {
    return "promoted_to_fraud_reports";
  }

  if (finalStatus === "duplicate_candidate") {
    return "reviewed_duplicate";
  }

  if (finalStatus === "needs_human_review") {
    return "needs_human_review";
  }

  if (finalStatus === "rejected") {
    return "rejected";
  }

  if (finalStatus === "ai_approved" || finalStatus === "ai_approved_after_redaction") {
    return "reviewed_ready_to_promote";
  }

  return "pending_review";
}

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  if (req.method !== "POST") {
    return json({ error: "Method not allowed" }, 405);
  }

  const cronSecret = Deno.env.get("CRON_SECRET");
const receivedCronSecret = req.headers.get("x-cron-secret");

if (!cronSecret || receivedCronSecret !== cronSecret) {
  return json({ ok: false, error: "Unauthorized cron request" }, 401);
}

  const supabaseUrl = Deno.env.get("SUPABASE_URL");
  const serviceKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY");

  if (!supabaseUrl || !serviceKey) {
    return json({ error: "Missing Supabase env vars" }, 500);
  }

  const supabase = createClient(supabaseUrl, serviceKey);

  let body: any = {};
  try {
    body = await req.json();
  } catch {
    body = {};
  }

  const limit = clampInt(body?.limit, 1, 10, 3);
  const dryRun = Boolean(body?.dryRun);

  const { data: reports, error: fetchError } = await supabase
    .from("fraud_chat_reports")
    .select("*")
    .eq("ai_review_status", "unreviewed")
    .eq("status", "pending_review")
    .order("created_at", { ascending: true })
    .limit(limit);

  if (fetchError) {
    return json({ error: "Failed to fetch reports", details: fetchError }, 500);
  }

  const results = [];

  for (const report of (reports || []) as FraudChatReport[]) {
    try {
      const similarMatches = await getSimilarMatches(supabase, report);
      const prompt = buildReviewPrompt(report, similarMatches);
      const review = await callOpenAIReview(prompt);

      let promotedFraudReportId: string | null = null;
      let promotionError: unknown = null;

      const canPromote =
        review.ai_promote_candidate &&
        (review.ai_review_status === "ai_approved" ||
          review.ai_review_status === "ai_approved_after_redaction") &&
        review.privacy_level !== "private_sensitive" &&
        !dryRun;

      if (canPromote) {
        const insertPayload = buildFraudReportInsert(report, review);
        const { data: inserted, error: insertError } = await supabase
          .from("fraud_reports")
          .insert(insertPayload)
          .select("id")
          .single();

        if (insertError) {
          promotionError = insertError;
        } else {
          promotedFraudReportId = inserted?.id ?? null;
        }
      }

      const finalStatus =
        promotedFraudReportId
          ? "ai_promoted"
          : review.ai_review_status;

            if (!dryRun) {
        const { error: updateError } = await supabase
          .from("fraud_chat_reports")
          .update({
            ai_review_status: finalStatus,
            ai_review_score: review.ai_review_score,
            ai_review_recommendation: review.ai_review_recommendation,
            ai_review_reason: review.ai_review_reason,
            ai_detected_sensitive_data: review.ai_detected_sensitive_data,
            ai_duplicate_likelihood: review.ai_duplicate_likelihood,
            ai_promote_candidate: review.ai_promote_candidate,
            ai_reviewed_at: new Date().toISOString(),
            ai_review_json: {
              ...review,
              similar_matches: similarMatches,
              dry_run: dryRun,
              promotion_error: promotionError,
            },
            promoted_fraud_report_id: promotedFraudReportId,
            promoted_at: promotedFraudReportId ? new Date().toISOString() : null,
            status: mapReviewStatusToChatReportStatus(finalStatus),
          })
          .eq("id", report.id);

        if (updateError) {
          results.push({
            id: report.id,
            ok: false,
            error: "update_error",
            details: updateError,
          });
          continue;
        }
      }

      results.push({
        id: report.id,
        ok: true,
        review_status: finalStatus,
        review_score: review.ai_review_score,
        duplicate_likelihood: review.ai_duplicate_likelihood,
        promote_candidate: review.ai_promote_candidate,
        promoted_fraud_report_id: promotedFraudReportId,
        dry_run: dryRun,
      });
    } catch (error) {
      console.error("review_report_error", report.id, error);

      await supabase
        .from("fraud_chat_reports")
        .update({
          ai_review_status: "needs_human_review",
          ai_review_reason: `Error en revisión automática: ${String(error)}`,
          ai_reviewed_at: new Date().toISOString(),
          ai_promote_candidate: false,
        })
        .eq("id", report.id);

      results.push({
        id: report.id,
        ok: false,
        error: String(error),
      });
    }
  }

  return json({
    ok: true,
    processed: results.length,
    dryRun,
    results,
  });
});
